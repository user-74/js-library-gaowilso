Examples using FunLoader.js

Head to Documentation

Each example uses a loading bar that adds 10% every second

const p = new ProgressBar(options)

Default Loading Bar

options = {}

Default Draggable Loading Bar

options = {draggable: true}

Custom Size Loading Bar

options = {height: 75, width: "100%", fontSize: 25}

Custom Background Image Loading Bar

options = {height: 50,
image: {source: "images/pic.jpg", leftShift:200, upShift:45 }}

Default Clicker Game Loading Bar

const p = new ClickerProgressBar(options)
options = {}

Custom Color and Opacity Loading Bar

options = {opacity: 0.5, gradient: ["#000000, #FFFFFF"]}

Overflow Loading Bar

options = {overflow: true, removeWhenDone: false}

Custom Draggable Clicker Game Loading Bar

options = {draggable: true,
clickPurchases: [{name: "God", cost: 3, rate: 300}, {name: "UofT", cost:300, rate: -300}]}

Custom Finished Message Loading Bar

options = {finishedMessage: "Loading Complete!",
removeWhenDone: false}
